# Программа для ввода этапов развития человека (тест по биологии)
# Студент вводит стадии, программа выводит их через "=>"
# Также программа показывает правильную последовательность для самопроверки

# Правильная последовательность (эталон)
correct_stages = [
    "Австралопитек",
    "Человек умелый (Homo habilis)",
    "Человек прямоходящий (Homo erectus)",
    "Человек разумный (Homo sapiens)",
    "Homo sapiens sapiens"   # последняя стадия по подсказке
]

print("=== Тест по биологии: этапы развития человека ===\n")
print("Введите свои варианты этапов в том порядке, который считаете правильным.")
print("(Для самопроверки в конце будет показан правильный ответ)\n")

# Запрашиваем количество этапов (студент может ввести столько, сколько знает)
count = int(input("Сколько этапов вы будете вводить? "))

user_stages = []
for i in range(1, count + 1):
    stage = input(f"Этап {i}: ")
    user_stages.append(stage)

# Вывод введённых студентом этапов
print("\nВаша последовательность:")
print(*user_stages, sep=" => ")

# Вывод правильной последовательности (эталон)
print("\nПравильная последовательность (для самопроверки):")
print(*correct_stages, sep=" => ")

print("\nПодсказка: последняя стадия — Homo sapiens sapiens.")
